  <script>
    // Acessibilidade e lógica das abas
    const tabs = document.querySelectorAll('[role="tab"]');
    const panels = document.querySelectorAll('[role="tabpanel"]');

    function openTab(panelId){
      panels.forEach(p=>{
        if(p.id===panelId){ p.hidden=false; } else { p.hidden=true; }
      });
      tabs.forEach(t=>{
        const controls = t.getAttribute('aria-controls');
        const target = controls || t.id.replace('tab-','panel-');
        t.setAttribute('aria-selected', (target===panelId).toString());
      });
      // foco para leitores de tela
      document.getElementById(panelId).focus?.();
    }

    tabs.forEach(t=>{
      t.addEventListener('click', ()=>{
        const panel = t.getAttribute('aria-controls');
        openTab(panel);
      });
      t.addEventListener('keydown', (e)=>{
        const curIndex = Array.from(tabs).indexOf(document.activeElement);
        if(e.key === 'ArrowRight') tabs[(curIndex+1)%tabs.length].focus();
        if(e.key === 'ArrowLeft') tabs[(curIndex-1+tabs.length)%tabs.length].focus();
      });
    });

    // Inicializa com a aba Início visível
    openTab('panel-home');

    // Atualiza ano no rodapé
    document.getElementById('year').textContent = new Date().getFullYear();

    function share(){
      const text = encodeURIComponent('Conheça os malefícios da xenofobia e como agir: Contra a Xenofobia');
      const url = encodeURIComponent(location.href);
      // tenta Web Share API
      if(navigator.share){
        navigator.share({title:document.title,text:url}).catch(()=>{});
      } else {
        window.open(`https://twitter.com/intent/tweet?text=${text}%20${url}`,'_blank');
      }
    }
  </script>
